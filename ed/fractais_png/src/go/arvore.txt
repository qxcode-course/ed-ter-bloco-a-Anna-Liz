package main

import (
	"fmt"
)

func arvore(pen *Pen, galho float64) {
	if galho < 2 {
		return
	}

	pen.Walk(galho)
	pen.Left(30)
	arvore(pen, galho*0.7)
	pen.Right(60)
	arvore(pen, galho*0.7)
	pen.Left(30)
	pen.Walk(-galho)

}

func main() {
	pen := NewPen(600, 500)
	pen.SetLineWidth(0.7)
	pen.SetHeading(90)
	pen.SetPosition(300, 500)
	arvore(pen, 150)
	pen.SavePNG("tree.png")
	fmt.Println("PNG file created successfully.")
}
