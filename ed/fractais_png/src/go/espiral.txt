package main

import (
	"fmt"
	"math/rand"
)

func espiral(pen *Pen, dist float64) {
	if dist < 5 {
		return
	}

	r := float64(rand.Intn(255))
	g := float64(rand.Intn(255))
	b := float64(rand.Intn(255))

	pen.SetRGB(r, g, b)

	pen.Walk(dist)
	pen.Right(90)
	espiral(pen, dist-5)

}

func main() {
	pen := NewPen(600, 500)
	//pen.SetLineWidth(0.7)
	pen.SetHeading(90)
	pen.SetPosition(100, 450)
	espiral(pen, 400)
	pen.SavePNG("espiral.png")
	fmt.Println("PNG file created successfully.")
}
